### Providers

`IconSettings`

```ts
import { NgModule } from '@angular/core';
import { IconModule } from '@anglify/icon';

@NgModule({
  imports: [
    IconModule.forRoot({
      defaultSet: 'mdi',
      svgIconSets: {},
      internalIcons: {},
      defaultSize: 'regular',
    }),
  ],
})
export class YourModule {
  // ...
}
```
